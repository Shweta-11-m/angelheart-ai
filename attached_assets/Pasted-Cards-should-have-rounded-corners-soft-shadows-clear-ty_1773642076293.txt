Most browser voice systems (Web Speech API) work like this:

if (event.results[i].isFinal) {
   save transcript

def calculate_health_score(sleep, stress, workload, mood):
    score = 100

    score -= stress * 5
    score -= workload * 3

    if sleep < 6:

    if mood == "stressed":
